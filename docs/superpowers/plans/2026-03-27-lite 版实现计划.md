"""Lite 版数据模型"""
from decimal import Decimal
from enum import Enum
from typing import List, Optional
from pydantic import BaseModel, Field


class Direction(str, Enum):
    LONG = "LONG"
    SHORT = "SHORT"


class KlineData(BaseModel):
    """K 线数据"""
    symbol: str
    timeframe: str
    timestamp: int  # 毫秒时间戳
    open: Decimal
    high: Decimal
    low: Decimal
    close: Decimal
    volume: Decimal
    is_closed: bool

    class Config:
        use_enum_values = True


class SignalResult(BaseModel):
    """信号结果"""
    symbol: str
    timeframe: str
    direction: Direction
    entry_price: Decimal
    stop_loss: Decimal
    big_trend: str  # "Bullish" | "Bearish"
    pinbar_quality: float  # 0~1

    def __str__(self) -> str:
        direction_text = "🟢 做多" if self.direction == Direction.LONG else "🔴 做空"
        return (
            f"【{self.symbol}】{direction_text}\n"
            f"周期：{self.timeframe}\n"
            f"入场价：${float(self.entry_price):,.2f}\n"
            f"止损价：${float(self.stop_loss):,.2f}\n"
            f"大周期趋势：{self.big_trend}\n"
            f"形态质量：{self.pinbar_quality:.0%}"
        )


class ExchangeConfig(BaseModel):
    """交易所配置"""
    name: str = "binance"
    api_key: str
    api_secret: str
    testnet: bool = False


class PinbarConfig(BaseModel):
    """Pinbar 参数配置"""
    min_wick_ratio: float = 0.6
    max_body_ratio: float = 0.3


class StrategyConfig(BaseModel):
    """策略配置"""
    ema_period: int = 60
    pinbar: PinbarConfig = Field(default_factory=PinbarConfig)


class NotificationConfig(BaseModel):
    """通知配置"""
    feishu_webhook: str


class LiteConfig(BaseModel):
    """Lite 版完整配置"""
    exchange: ExchangeConfig
    symbols: List[str]
    timeframes: List[str]
    strategy: StrategyConfig = Field(default_factory=StrategyConfig)
    notification: NotificationConfig

    class Config:
        use_enum_values = True
EOF
```

"""Lite 版策略逻辑 - Pinbar + 顺大逆小"""
from decimal import Decimal
from typing import Optional
from models import KlineData, SignalResult, Direction


def calculate_wick_ratio(kline: KlineData) -> tuple[Decimal, Decimal, Decimal]:
    """
    计算影线占比

    Returns:
        (upper_wick_ratio, lower_wick_ratio, body_ratio)
    """
    high = kline.high
    low = kline.low
    open_price = kline.open
    close_price = kline.close

    # K 线总长度
    total_range = high - low
    if total_range == 0:
        return Decimal("0"), Decimal("0"), Decimal("0")

    # 实体长度
    body = abs(close_price - open_price)

    # 上影线长度
    upper_wick = high - max(open_price, close_price)

    # 下影线长度
    lower_wick = min(open_price, close_price) - low

    upper_wick_ratio = upper_wick / total_range
    lower_wick_ratio = lower_wick / total_range
    body_ratio = body / total_range

    return upper_wick_ratio, lower_wick_ratio, body_ratio


def is_bullish_pinbar(kline: KlineData, min_wick: float = 0.6, max_body: float = 0.3) -> bool:
    """
    检测看涨 Pinbar（长下影线，实体在顶部）

    条件:
    - 下影线占比 >= 60%
    - 实体占比 <= 30%
    - 实体位于 K 线顶部 10% 区间
    """
    upper_wick_ratio, lower_wick_ratio, body_ratio = calculate_wick_ratio(kline)

    # 下影线足够长
    if lower_wick_ratio < min_wick:
        return False

    # 实体足够小
    if body_ratio > max_body:
        return False

    # 实体位于顶部（收盘价接近最高价）
    body_top = max(kline.open, kline.close)
    body_bottom = min(kline.open, kline.close)
    total_range = kline.high - kline.low

    if total_range == 0:
        return False

    # 实体顶部距离最高价的距离 <= 10% 总长度
    upper_gap = kline.high - body_top
    upper_gap_ratio = upper_gap / total_range

    return upper_gap_ratio <= 0.1


def is_bearish_pinbar(kline: KlineData, min_wick: float = 0.6, max_body: float = 0.3) -> bool:
    """
    检测看跌 Pinbar（长上影线，实体在底部）

    条件:
    - 上影线占比 >= 60%
    - 实体占比 <= 30%
    - 实体位于 K 线底部 10% 区间
    """
    upper_wick_ratio, lower_wick_ratio, body_ratio = calculate_wick_ratio(kline)

    # 上影线足够长
    if upper_wick_ratio < min_wick:
        return False

    # 实体足够小
    if body_ratio > max_body:
        return False

    # 实体位于底部（开盘价接近最低价）
    body_top = max(kline.open, kline.close)
    body_bottom = min(kline.open, kline.close)
    total_range = kline.high - kline.low

    if total_range == 0:
        return False

    # 实体底部距离最低价的距离 <= 10% 总长度
    lower_gap = body_bottom - kline.low
    lower_gap_ratio = lower_gap / total_range

    return lower_gap_ratio <= 0.1


def get_ema_trend(close: Decimal, ema: Decimal) -> str:
    """
    根据收盘价与 EMA 的关系判断趋势

    Returns:
        "Bullish" | "Bearish" | "Neutral"
    """
    if close > ema:
        return "Bullish"
    elif close < ema:
        return "Bearish"
    else:
        return "Neutral"


def calculate_stop_loss(kline: KlineData, direction: Direction) -> Decimal:
    """
    计算止损价

    - LONG: Pinbar 最低价下方一点
    - SHORT: Pinbar 最高价上方一点
    """
    if direction == Direction.LONG:
        # 止损设在 Pinbar 最低价下方 0.1%
        return kline.low * Decimal("0.999")
    else:
        # 止损设在 Pinbar 最高价上方 0.1%
        return kline.high * Decimal("1.001")


def check_pinbar_signal(
    kline_15m: KlineData,
    context: dict
) -> Optional[SignalResult]:
    """
    检查是否产生 Pinbar 信号（顺大逆小逻辑）

    Args:
        kline_15m: 15 分钟 K 线
        context: 上下文信息，包含:
            - ema_1h: 1h 周期的 EMA 值
            - close_1h: 1h 周期的收盘价

    Returns:
        SignalResult if signal detected, None otherwise
    """
    ema_1h = context.get("ema_1h")
    close_1h = context.get("close_1h")

    if ema_1h is None or close_1h is None:
        return None

    # 1h 大周期趋势判定
    big_trend = get_ema_trend(close_1h, ema_1h)

    # 顺大逆小：只做顺势方向的 Pinbar
    if big_trend == "Bullish":
        # 多头趋势：只做看涨 Pinbar
        if not is_bullish_pinbar(kline_15m):
            return None

        direction = Direction.LONG
        entry_price = kline_15m.close
        stop_loss = calculate_stop_loss(kline_15m, direction)

        # 计算形态质量（下影线占比）
        _, lower_wick_ratio, _ = calculate_wick_ratio(kline_15m)

        return SignalResult(
            symbol=kline_15m.symbol,
            timeframe=kline_15m.timeframe,
            direction=direction,
            entry_price=entry_price,
            stop_loss=stop_loss,
            big_trend=big_trend,
            pinbar_quality=float(lower_wick_ratio)
        )

    elif big_trend == "Bearish":
        # 空头趋势：只做看跌 Pinbar
        if not is_bearish_pinbar(kline_15m):
            return None

        direction = Direction.SHORT
        entry_price = kline_15m.close
        stop_loss = calculate_stop_loss(kline_15m, direction)

        # 计算形态质量（上影线占比）
        upper_wick_ratio, _, _ = calculate_wick_ratio(kline_15m)

        return SignalResult(
            symbol=kline_15m.symbol,
            timeframe=kline_15m.timeframe,
            direction=direction,
            entry_price=entry_price,
            stop_loss=stop_loss,
            big_trend=big_trend,
            pinbar_quality=float(upper_wick_ratio)
        )

    # Neutral 趋势不产生信号
    return None
EOF

import asyncio

def format_signal_message(signal: SignalResult) -> str:
    """
    格式化信号通知消息

    Returns:
        Markdown 格式的消息文本
    """
    # 方向图标和文字
    if signal.direction == Direction.LONG:
        direction_icon = "🟢"
        direction_text = "做多"
        trend_text = "1h EMA 多头"
    else:
        direction_icon = "🔴"
        direction_text = "做空"
        trend_text = "1h EMA 空头"

    # 格式化价格
    entry_price = f"${float(signal.entry_price):,.2f}"
    stop_loss = f"${float(signal.stop_loss):,.2f}"

    # 形态质量百分比
    quality_percent = f"{signal.pinbar_quality:.0%}"

    message = (
        f"🐶 盯盘狗 - Pinbar 信号\n\n"
        f"币种：{signal.symbol}\n"
        f"周期：{signal.timeframe}\n"
        f"方向：{direction_icon} {direction_text}\n"
        f"入场价：{entry_price}\n"
        f"止损价：{stop_loss}\n\n"
        f"大周期趋势：{trend_text}\n"
        f"形态质量：标准 Pinbar ({quality_percent})\n\n"
        f"---\n"
        f"盯盘狗 Lite v0.1.0"
    )

    return message


async def send_feishu_notification(
    signal: SignalResult,
    webhook_url: str,
    timeout: int = 10
) -> bool:
    """
    发送飞书通知

    Args:
        signal: 信号结果
        webhook_url: 飞书 Webhook URL
        timeout: 请求超时（秒）

    Returns:
        True if sent successfully, False otherwise
    """
    message = format_signal_message(signal)

    # 飞书消息格式
    payload = {
        "msg_type": "text",
        "content": {
            "text": message
        }
    }

    try:
        async with aiohttp.ClientSession() as session:
            async with session.post(
                webhook_url,
                json=payload,
                timeout=aiohttp.ClientTimeout(total=timeout)
            ) as response:
                if response.status == 200:
                    return True
                else:
                    # 记录错误但不抛异常
                    print(f"[Notifier] 飞书推送失败，状态码：{response.status}")
                    return False

    except asyncio.TimeoutError:
        print("[Notifier] 飞书推送超时")
        return False

    except Exception as e:
        print(f"[Notifier] 飞书推送异常：{e}")
        return False
EOF

#!/usr/bin/env python3
"""
盯盘狗 Lite v0.1.0 - 个人盯盘工具

极简版 Pinbar 策略监控，支持 Docker 一键部署
"""
import asyncio
import logging
import yaml
from pathlib import Path
from decimal import Decimal
from typing import Dict, Any

import ccxt.async_support as ccxt
from models import LiteConfig, ExchangeConfig, KlineData, SignalResult
from strategy import check_pinbar_signal
from notifier import send_feishu_notification


# ==================== 配置 ====================

CONFIG_FILE = "config.yaml"
LOG_FILE = "logs/lite.log"
LOG_FORMAT = "%(asctime)s %(levelname)s  %(message)s"
DATE_FORMAT = "%Y-%m-%d %H:%M:%S"

# ==================== 日志 ====================


def setup_logging():
    """配置日志系统"""
    # 确保 logs 目录存在
    Path("logs").mkdir(exist_ok=True)

    # 文件处理器
    file_handler = logging.FileHandler(LOG_FILE, encoding="utf-8")
    file_handler.setLevel(logging.INFO)
    file_handler.setFormatter(logging.Formatter(LOG_FORMAT, DATE_FORMAT))

    # 终端处理器
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.INFO)
    console_handler.setFormatter(logging.Formatter(LOG_FORMAT, DATE_FORMAT))

    # 根日志器
    logger = logging.getLogger()
    logger.setLevel(logging.INFO)
    logger.addHandler(file_handler)
    logger.addHandler(console_handler)

    return logger


# ==================== 配置加载 ====================


def load_config(config_path: str = CONFIG_FILE) -> LiteConfig:
    """加载配置文件"""
    with open(config_path, "r", encoding="utf-8") as f:
        config_dict = yaml.safe_load(f)

    return LiteConfig(**config_dict)


# ==================== 交易所网关 ====================


class ExchangeGateway:
    """Binance 交易所网关"""

    def __init__(self, config: ExchangeConfig):
        self.config = config
        self.exchange: ccxt.binance = None

    async def connect(self):
        """连接交易所"""
        self.exchange = ccxt.binance({
            "apiKey": self.config.api_key,
            "secret": self.config.api_secret,
            "enableRateLimit": True,
            "options": {
                "defaultType": "future",  # 合约
                "test": self.config.testnet,
            }
        })

        # 加载市场数据
        await self.exchange.load_markets()
        logging.info(f"已连接 Binance {'测试网' if self.config.testnet else '实盘'}")

    async def subscribe_klines(
        self,
        symbols: list,
        timeframes: list,
        callback
    ):
        """
        订阅 K 线数据

        Args:
            symbols: 币种列表，如 ["BTC/USDT:USDT"]
            timeframes: 周期列表，如 ["15m", "1h"]
            callback: 回调函数 on_kline(kline: KlineData)
        """
        # 为每个币对的每个周期订阅
        for symbol in symbols:
            for timeframe in timeframes:
                asyncio.create_task(
                    self._watch_kline(symbol, timeframe, callback)
                )

    async def _watch_kline(
        self,
        symbol: str,
        timeframe: str,
        callback
    ):
        """监控单个币对的 K 线"""
        while True:
            try:
                # 获取最新 K 线
                klines = await self.exchange.fetch_ohlcv(
                    symbol, timeframe, limit=5
                )

                for k in klines:
                    kline = KlineData(
                        symbol=symbol,
                        timeframe=timeframe,
                        timestamp=k[0],
                        open=Decimal(str(k[1])),
                        high=Decimal(str(k[2])),
                        low=Decimal(str(k[3])),
                        close=Decimal(str(k[4])),
                        volume=Decimal(str(k[5])),
                        is_closed=False  # WebSocket 模式下可改为 True
                    )
                    await callback(kline)

                # 轮询间隔
                await asyncio.sleep(10)

            except Exception as e:
                logging.error(f"[{symbol}] K 线监控异常：{e}")
                await asyncio.sleep(5)

    async def get_ema(self, symbol: str, timeframe: str, period: int) -> Decimal:
        """获取 EMA 值（简化实现：用 SMA 近似）"""
        klines = await self.exchange.fetch_ohlcv(symbol, timeframe, limit=period)
        closes = [Decimal(str(k[4])) for k in klines]
        return sum(closes) / len(closes)

    async def close(self):
        """关闭连接"""
        if self.exchange:
            await self.exchange.close()


# ==================== 信号管道 ====================


class SignalPipeline:
    """Lite 版信号处理管道"""

    def __init__(self, config: LiteConfig):
        self.config = config
        self.gateway: ExchangeGateway = None
        self.kline_context: Dict[str, Any] = {}  # 存储 1h K 线上下文
        self.last_signal_time: Dict[str, int] = {}  # 冷却缓存

    def set_gateway(self, gateway: ExchangeGateway):
        self.gateway = gateway

    async def on_kline(self, kline: KlineData):
        """K 线数据回调"""
        logging.debug(
            f"[{kline.symbol}] {kline.timeframe} "
            f"O:{kline.open} H:{kline.high} L:{kline.low} C:{kline.close}"
        )

        # 1h K 线用于判断大周期趋势
        if kline.timeframe == "1h":
            self._update_1h_context(kline)
            return

        # 15m K 线用于检测 Pinbar
        if kline.timeframe == "15m" and kline.is_closed:
            await self._check_15m_signal(kline)

    def _update_1h_context(self, kline: KlineData):
        """更新 1h 上下文（用于 EMA 计算）"""
        key = f"{kline.symbol}:1h"
        self.kline_context[key] = {
            "close_1h": kline.close,
            "timestamp": kline.timestamp
        }

    async def _check_15m_signal(self, kline: KlineData):
        """检查 15m Pinbar 信号"""
        # 冷却检查（5 分钟内不重复推送同一币种）
        if not self._check_cooldown(kline.symbol):
            return

        # 获取 1h 上下文
        context_key = f"{kline.symbol}:1h"
        context_1h = self.kline_context.get(context_key)

        if not context_1h:
            logging.warning(f"[{kline.symbol}] 缺少 1h 数据，跳过信号检测")
            return

        # 获取 EMA
        try:
            ema_1h = await self.gateway.get_ema(
                kline.symbol, "1h", self.config.strategy.ema_period
            )
        except Exception as e:
            logging.error(f"[{kline.symbol}] EMA 获取失败：{e}")
            return

        # 构建策略上下文
        strategy_context = {
            "ema_1h": ema_1h,
            "close_1h": context_1h["close_1h"]
        }

        # 检测信号
        signal = check_pinbar_signal(kline, strategy_context)

        if signal:
            logging.info(
                f"[{kline.symbol}] 🟢 做多" if signal.direction.value == "LONG"
                else f"[{kline.symbol}] 🔴 做空"
                f" - 入场价：{signal.entry_price}"
            )

            # 推送通知
            await self._notify_signal(signal)

            # 更新冷却缓存
            self._update_cooldown(kline.symbol)

    def _check_cooldown(self, symbol: str) -> bool:
        """检查冷却时间（5 分钟）"""
        import time
        current_time = int(time.time())
        last_time = self.last_signal_time.get(symbol, 0)
        return (current_time - last_time) > 300  # 5 分钟

    def _update_cooldown(self, symbol: str):
        """更新冷却缓存"""
        import time
        self.last_signal_time[symbol] = int(time.time())

    async def _notify_signal(self, signal: SignalResult):
        """发送信号通知"""
        webhook_url = self.config.notification.feishu_webhook
        success = await send_feishu_notification(signal, webhook_url)

        if success:
            logging.info("[通知] 飞书推送成功")
        else:
            logging.error("[通知] 飞书推送失败")


# ==================== 主程序 ====================


async def main():
    """主函数"""
    # 设置日志
    setup_logging()
    logging.info("=" * 60)
    logging.info("盯盘狗 Lite v0.1.0 - 启动")
    logging.info("=" * 60)

    try:
        # 加载配置
        logging.info("Phase 1: 加载配置...")
        config = load_config()
        logging.info(f"监控币种：{', '.join(config.symbols)}")
        logging.info(f"监控周期：{', '.join(config.timeframes)}")

        await gateway.connect()

        # 创建信号管道

        )

        # 运行
        logging.info("=" * 60)
        logging.info("SYSTEM READY - 开始监控")
        logging.info("=" * 60)

        # 保持运行
        while True:
            await asyncio.sleep(60)

    except KeyboardInterrupt:
        logging.info("收到中断信号，正在关闭...")
    except Exception as e:
        logging.error(f"运行异常：{e}", exc_info=True)
    finally:
        if "gateway" in locals():
            await gateway.close()
        logging.info("盯盘狗 Lite - 已停止")

## 策略逻辑

### 顺大逆小 + Pinbar

1. **顺大**: 1h 周期 EMA 判断大趋势
   - 收盘价 > EMA → 多头趋势
   - 收盘价 < EMA → 空头趋势

2. **逆小**: 15m 周期检测 Pinbar 形态
   - 多头趋势：只做看涨 Pinbar（长下影线）
   - 空头趋势：只做看跌 Pinbar（长上影线）

### Pinbar 判定规则

| 参数 | 默认值 | 说明 |
|------|--------|------|
| 影线占比 | ≥60% | 下影线/上影线长度占比 |
| 实体占比 | ≤30% | 实体长度占比 |
| 实体位置 | ≤10% | 实体位于 K 线顶部/底部区间 |

## 常见问题

### Q: API Key 需要什么权限？
A: **只读权限**即可。系统仅用于行情监控，不会进行交易。

### Q: 如何获取飞书 Webhook URL？
A: 在飞书群聊中添加"自定义机器人"，复制 Webhook 地址。

### Q: 支持哪些交易所？
A: 当前仅支持 Binance 合约。

### Q: 如何修改策略参数？
A: 编辑 `config.yaml` 后重启容器：
```bash
docker-compose restart
```
